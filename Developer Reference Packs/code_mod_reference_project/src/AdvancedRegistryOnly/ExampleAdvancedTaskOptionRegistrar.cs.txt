using NurtaleNesche.Modding.Registries;

namespace ExampleMods.AdvancedRegistryOnly
{
    // Advanced/registry-only example.
    // Use only after reading Tutorial Group 5 and understanding task option categories.
    public sealed class ExampleAdvancedTaskOptionRegistrar
    {
        public void RegisterSolo(Patroller.SoloActionTaskOptionRegistry registry)
        {
            registry.RegisterSoloActionTaskOption<ExampleSoloActionTaskOption>(
                "task_option.solo_action.example_noop",
                allowOverride: false);
        }

        public void RegisterChase(Patroller.ChaseTaskOptionRegistry registry)
        {
            registry.RegisterChaseTaskOption<ExampleChaseTaskOption>(
                "task_option.chase.example_noop",
                allowOverride: false);
        }

        public void RegisterAbility(Patroller.AbilityTaskOptionRegistry registry)
        {
            registry.RegisterAbilityTaskOption<ExampleAbilityTaskOption>(
                "task_option.ability.example_noop",
                allowOverride: false);
        }

        public void RegisterCollectCaptive(Patroller.CollectCaptiveTaskOptionRegistry registry)
        {
            registry.RegisterCollectCaptiveTaskOption<ExampleCollectCaptiveTaskOption>(
                "task_option.collect_captive.example_noop",
                allowOverride: false);
        }
    }
}
